package org.jpmml.evaluator.visitors;

import java.util.HashMap;

import org.dmg.pmml.PMMLObject;

abstract
class ElementHashMap<V extends PMMLObject> extends HashMap<ElementKey, V> {

	abstract
	public ElementKey createKey(V object);

	public V intern(V object){
		ElementKey key = createKey(object);

		V result = get(key);

		if(result == null){
			result = object;

			put(key, result);
		}

		return result;
	}
}
